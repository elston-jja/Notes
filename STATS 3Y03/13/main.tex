\section{Joint Probability Distributions}

The \b{joint probability mass function} of discrete random variables $X$ and $Y$, denoted as $f_{XY}(x,y)$ satisfies
\begin{enumerate}
	\item $f_{XY} (x,y) \geq 0 $
	\item $\sum_X \sum_Y f_{XY}(x,y) = 1$
	\item $f_{XY} = P(X = x, Y = y)$
	\end{enumerate}
 The \b{joint probability density function} for the continuous random variables $X$ and $Y$, denoted as $f_{XY}(x,y)$, satisfies the following properties:
	 \begin{enumerate}
 	\item $f_{XY}(x,y) \geq 0 $ for all $x,y$
 	\item $\int_{-\infty}^{\infty} \int_{-\infty}^{\infty} f_{XY}(x,y) \, dx\, dy = 1$
	\item For any region $R$ of a two-dimensional space,
	\begin{align}
	P((X,Y) \in R) = \iint_R f_{XY}(x,y) dx\, dy
	\end{align}
	\end{enumerate}

	\de{
	\b{Example 5.2} Joint p.d.f of $(X,Y)$ is
	\begin{align*}
	f(x,y) = 
	\begin{cases}
	 6(10^{-6}) \exp{-0.001x -0.002y} & 0< x < y < \infty\\
	 0 & \text{Else} 
	\end{cases}
	\end{align*}
	Find $P(X < 1000, Y < 2000)$	
	\\
	
	\b{Ans.}
	\begin{align*}
	6(10^{-6}) \int_{x=0}^{1000} \bigg(\int_{y=x}^{2000} e^{-0.002y} \, dy \bigg) e^{-0.001x} dx = 0.9155
	\end{align*}
